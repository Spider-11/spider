#include<stdio.h>
#include<string.h>
int bubble(char ar[][20], int size, int index) {
	int j;
    if (index == size - 1) {
        return 0;
    }
    for (j = 0; j < size - index - 1; j++) {
        if (strcmp(ar[j], ar[j + 1]) > 0) {
            char temp[100];
            strcpy(temp, ar[j]);
            strcpy(ar[j], ar[j + 1]);
            strcpy(ar[j + 1], temp);
        }
    }
    return bubble(ar, size, index + 1);
}
int main() {
    char ar[5][20];
    int i;
    for (i = 0; i < 5; i++) {
    	printf("Enter the string at %d location\t",i+1);
        scanf("%s", ar[i]);
    }
    bubble(ar, 5, 0);
    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", ar[i]);
    }
}
