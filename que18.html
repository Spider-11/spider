#include<stdio.h>
void reversek(int ar[],int size,int k)
{
int i;
for(i=k;i>=0;i--)
{
printf("Reverse order is %d\n",ar[i]);
}
for(i=k+1;i<size;i++)
{
printf("Reverse order is %d\n",ar[i]);
}
}
int main()
{
int ar[5],i,k;
printf("Enter the arrays\n");
for(i=0;i<5;i++)
{
printf("Enter the array at location %d\t",i+1);
scanf("%d",&ar[i]);
}
printf("Enter the number of array to reverse\n");
scanf("%d",&k);
reversek(ar,5,k);
}
