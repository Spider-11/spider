#include<stdio.h>
    void insertion(int ar[],int size)
    {
    int i,j,temp;
    for(i=1;i<size;i++)
    {
    temp=ar[i];
    j=i-1;
    while(temp<ar[j] && j>=0)
    {
    ar[j+1]=ar[j];
    j--;
    }
    ar[j+1]=temp;
    }
    }
    int main()

    int ar[10],i;

    scanf("%d",&ar[i]);

    insertion(ar,10);
    printf("Sorted Arrays are\n");

    {
    printf("%d\n",ar[i]);
    }
    }
