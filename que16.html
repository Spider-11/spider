#include<stdio.h>
void nextsmaller(int arr[], int n)
{
	int next, i, j;
	for (i=0; i<n; i++)
	{
		next = -1;
		for (j = i+1; j<n; j++)
		{
			if (arr[i] > arr[j])
			{
				next = arr[j];
				break;
			}
		}
		printf("Next smaller element is %d\n",next);
	}
}

int main()
{
	int arr[5],i;
	printf("Enter the arrays\n");
	for(i=0;i<5;i++)
	{
	printf("Enter the array at location %d\t",i+1);
	scanf("%d",&arr[i]);
	}
	nextsmaller(arr, 5);
}
