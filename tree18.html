#include<stdio.h>
#include<stdlib.h>

struct Node {
    int data;
    struct Node* left;
    struct Node* right;
};

struct Node* createNewNode(int data) {
    struct Node* node = (struct Node*)malloc(sizeof(struct Node));
    node->data = data;
    node->left = node->right = NULL;
    return node;
}
void printTree(struct Node* root) {
    if (root == NULL) {
        return;
    }
    printTree(root->left);
    printf("%d ", root->data);
    printTree(root->right);
}

int main() {
    int data;
    struct Node* root = NULL;
    struct Node* newNode = NULL;
    struct Node* currentNode = NULL;
    int level = 0;
    printf("Enter the root node value: ");
    scanf("%d", &data);
    root = createNewNode(data);
    currentNode = root;
    int choice;
    do {
        printf("Add a child node (1: left, 2: right, 0: no child): ");
        scanf("%d", &choice);
        if (choice == 0) {
            printf("Enter the level for the new node (starting from 1): ");
            scanf("%d", &level);
            currentNode = root;
        } else if (choice == 1 || choice == 2) {
            printf("Enter the node value: ");
            scanf("%d", &data);
            int bit = (data >> (level - 1)) & 1;
            if (bit == 0) {
                if (currentNode->left == NULL) {
                    newNode = createNewNode(data);
                    currentNode->left = newNode;
                } else {
                    currentNode = currentNode->left;
                }
            } else {
                if (currentNode->right == NULL) {
                    newNode = createNewNode(data);
                    currentNode->right = newNode;
                } else {
                    currentNode = currentNode->right;
                }
            }
        } else {
            printf("Invalid choice, please enter 1, 2, or 0.\n");
        }
        printf("Binary Tree (In-order traversal): ");
        printTree(root);
        printf("\n");
    } while (1);
    return 0;
}
