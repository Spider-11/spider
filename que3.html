#include<stdio.h>
    #include<stdlib.h>
    struct que
    {
    int data;
    struct que *next;
    };
    struct que *rear=NULL,*front=NULL,*newnode,*temp;
    int insert(int count)
    {
    if(rear==NULL)
    {
    newnode=(struct que*) malloc(sizeof(struct que));
    printf("Enter the data\n");
    scanf("%d",&(newnode->data));
    newnode->next=NULL;
    rear=front=newnode;
    count++;
    }
    else
    {
    newnode=(struct que*) malloc(sizeof(struct que));
    printf("Enter the data\n");
    scanf("%d",&(newnode->data));
    newnode->next=NULL;
    while(rear->next!=NULL)
    {
    rear=rear->next;
    }
    rear->next=newnode;
    count++;
    }
    return count;
    }
    int del(int count)
    {
    if(front==NULL)
    {
    printf("Queue is empty\n");
    }
    else
    {
    temp=front;
    printf("Deleted data in queue is %d",temp->data);
    front=front->next;
    free(temp);
    count--;
    }
    return count;
    }
    void peek()
    {
    if(rear==NULL)
    {
    printf("Queue is empty\n");
    }
    else
    {
    rear=rear->next;
    printf("Top element in queue is %d",rear->data);
    }
    }
    void isempty()
    {
    if(rear==NULL)
    {
    printf("Queue is empty\n");
    }
    else
    {
    printf("Queue is not empty\n");
    }
    }
    void isfull()
    {
    if(front==rear)
    {
    printf("Queue is full\n");
    }
    else
    {
    printf("Queue is not full\n");
    }
    }
    int main()
    {
    int ch,count=0;
    while(1)
    {
    printf("\n1. Insert\n");
    printf("2. Delete\n");
    printf("3. Peek\n");
    printf("4. Isempty\n");
    printf("5. Isfull\n");
    printf("6. Total nodes\n");
    printf("7. Exit\n");
    printf("Enter your choice\n");
    scanf("%d",&ch);
    switch(ch)
    {
    case 1:
    count=insert(count);
    break;
    case 2:
    count=del(count);
    break;
    case 3:
    peek();
    break;
    case 4:
    isempty();
    break;
    case 5:
    isfull();
    break;
    case 6:
    printf("Total nodes in queue is %d",count);
    break;
    case 7:
    exit(1);
    default:
    printf("Enter the valid number\n");
    }
    }
    }
