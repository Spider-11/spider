#include <stdio.h>
#include <string.h>
int longeststring(char* str) {
    int n = strlen(str);
    int stack[10], top = -1;
    int result = 0;
    stack[++top] = -1;
    for (int i = 0; i < n; i++) {
        if (str[i] == '(') {
            stack[++top] = i;
        } else {
            if (top != -1) {
                top--;
       if (top != -1) {
	  int current_length = i - stack[top];
        result = (current_length > result) ? current_length : result
            } else {
                stack[++top] = i;
                }
            } else {
         stack[++top] = i;
            }
        }
    }
    return result;
}

int main() {
    char str[10];
    printf("Enter a string containing parentheses: ");
    scanf("%s", str);

    int result = longeststring(str);
    printf("Length of the longest valid string: %d\n", result);

}
