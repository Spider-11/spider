#include<stdio.h>
int selection(int ar[], int size, int index) {
	int j,temp;
    if (index == size - 1) {
        return 0;
    }
    for ( j = index + 1; j < size; j++) {
        if (ar[index] > ar[j]) {
            temp = ar[index];
            ar[index] = ar[j];
            ar[j] = temp;
        }
    }
    return selection(ar, size, index + 1);
}
int main() {
    int ar[10], i;
    printf("Enter the arrays\n");
    for (i = 0; i < 10; i++) {
    	printf("Enter the array at location %d\t",i+1);
        scanf("%d", &ar[i]);
    }
    selection(ar, 10, 0);
    printf("Sorted Arrays are\n");
    for (i = 0; i < 10; i++) {
        printf("%d\n", ar[i]);
    }
}
