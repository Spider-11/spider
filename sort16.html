#include<stdio.h>
#include<string.h>
int insertion(char ar[][10], int size, int index) {
	char temp[10];
	int j;
    if (index >= size) {
        return 0;
    }
    strcpy(temp, ar[index]);
    j = index - 1;
    while (j >= 0 && strcmp(ar[j], temp) > 0) {
        strcpy(ar[j + 1], ar[j]);
        j--;
    }
    strcpy(ar[j + 1], temp);
    return insertion(ar, size, index + 1);
}

int main() {
    char ar[5][10];
    int i;
    for (i = 0; i < 5; i++) {
    	printf("Enter the string at %d location\t",i+1);
        scanf("%s", ar[i]);
    }
    insertion(ar, 5, 0);
    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", ar[i]);
    }
}
