#include<stdio.h>
    #include<stdlib.h>
    struct stack
    {
    int data;
    struct stack *next;
    };
    struct stack *top=NULL,*temp,*newnode;
    int push(int count)
    {
    if(top==NULL)
    {
    newnode=(struct stack*) malloc(sizeof(struct stack));
    printf("Enter the data\n");
    scanf("%d",&(newnode->data));
    newnode->next=NULL;
    top=newnode;
    count++;
    }
    else
    {
    newnode=(struct stack*) malloc(sizeof(struct stack));
    printf("Enter the data\n");
    scanf("%d",&(newnode->data));
    newnode->next=top;
    top=newnode;
    count++;
    }
    return count;
    }
    int pop(int count)
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    temp=top;
    printf("data from the top of the stack is %d\n",temp->data);
    top=top->next;
    free(temp);
    count--;
    }
    return count;
    }
    void peek()
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    printf("Data from the top of the stack is %d\n",top->data);
    }
    }
    void isempty()
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    printf("Stack is not empty\n");
    }
    }
    void isfull(int count)
    {
    if(count==count+1)
    {
    printf("Stack is full\n");
    }
    else
    {
    printf("Stack is not full\n");
    }
    }
