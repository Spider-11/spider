#include<stdio.h>
#include<string.h>

void quick_sort(char a[][10], int low, int high);
int partition(char a[][10], int low, int high);

void quick_sort(char a[][10], int low, int high) {
    int pivot;
    if (low < high) {
        pivot = partition(a, low, high);
        quick_sort(a, low, pivot - 1);
        quick_sort(a, pivot + 1, high);
    }
}

int partition(char a[][10], int low, int high) {
    char pivot[10];
    strcpy(pivot, a[high]);
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (strcmp(a[j], pivot) <= 0) {
            i++;
            char temp[10];
            strcpy(temp, a[i]);
            strcpy(a[i], a[j]);
            strcpy(a[j], temp);
        }
    }
    char temp[10];
    strcpy(temp, a[i + 1]);
    strcpy(a[i + 1], a[high]);
    strcpy(a[high], temp);
    return (i + 1);
}

int main() {
    int i;
    char a[5][10];
    for (i = 0; i < 5; i++) {
        printf("Enter the string at location %d: ", i + 1);
        scanf("%s", a[i]);
    }

    quick_sort(a, 0, 4);

    printf("Sorted array:\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", a[i]);
    }
}
