#include<stdio.h>
#include<string.h>

void insertion(char str[][10], int size) {
    int i, j;
    char temp[10];

    for (i = 1; i < size; i++) {
        strcpy(temp, str[i]);
        j = i - 1;

        while (j >= 0 && strcmp(temp, str[j]) < 0) {
            strcpy(str[j + 1], str[j]);
            j--;
        }

        strcpy(str[j + 1], temp);
    }
}

int main() {
    char str[5][10];
    int i;

    printf("Enter the strings\n");
    for (i = 0; i < 5; i++) {
        printf("Enter the string at location %d: ", i + 1);
        scanf("%s", str[i]);
    }

    insertion(str, 5);

    printf("Sorted Strings are\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", str[i]);
    }

}
