#include<stdio.h>
int findmin(int arr[], int size) {
	int sum;
    if (size == 1) {
        return arr[0];
    } else {
        sum = findmin(arr, size - 1);
        return (sum < arr[size - 1]) ? sum : arr[size - 1];
    }
}

int main() {
    int arr[5],i,ele;
    printf("Enter the arrays:\n");
    for (i = 0; i < 5; ++i) {
    	printf("Enter the array at location %d\t",i+1);
        scanf("%d", &arr[i]);
    }
    ele = findmin(arr,5);
    printf("The smallest element in the array is: %d\n", ele);
}
