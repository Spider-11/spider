#include <stdio.h>
int binary(char str[], char ch, int length) {
    int low = 0,mid;
    int high = length - 1;
    while (low <= high) {
         mid = (low + high) / 2;
        if (str[mid] == ch) {
            return mid+1;
        } else if (str[mid] < ch) {
            low = mid + 1; 
        } else {
            high = mid - 1; 
        }
    }
    return -1; 
}
int main() {
    char str[10],ch;
    int length,result;
    printf("Enter a sorted string: ");
    scanf("%s", str);
    printf("Enter a character to search: ");
    scanf(" %c", &ch);
    length = 0;
    while (str[length] != '\0') {
        length++;
    }
    result = binary(str, ch, length);
    if (result != -1) {
        printf("Character '%c' found at index %d", ch, result);
    } else {
        printf("Character '%c' not found in the string", ch);
    }
}
