#include<stdio.h>

int findmax(int arr[], int size) {
	int sum;
    if (size == 1) {
        return arr[0];
    } else {
        sum = findmax(arr, size - 1);
        
        return (sum > arr[size - 1]) ? sum : arr[size - 1];
    }
}
