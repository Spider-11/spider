#include<stdio.h>
void nextgreater(int arr[], int n) 
{ 
	int next, i, j; 
	for (i=0; i<n; i++) 
	{ 
		next = -1; 
		for (j = i+1; j<n; j++) 
		{ 
			if (arr[i] < arr[j]) 
			{ 
				next = arr[j]; 
				break; 
			} 
		} 
		printf("Next greater element is %d\n", next); 
	} 
} 

int main() 
{ 
	int arr[5],i;
	printf("Enter the arrays");
	for(i=0;i<5;i++)
	{
	printf("Enter the array at location %d\t",i+1);
	scanf("%d",&arr[i]);
	 } 
	nextgreater(arr, 5);  
}
