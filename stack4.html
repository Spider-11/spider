#include<stdio.h>
    #include<stdlib.h>
    struct stack
    {
    char ch;
    struct stack *next;
    };
    struct stack *top=NULL,*temp,*newnode;
    int push(int count)
    {
    fflush(stdin);
    if(top==NULL)
    {
    newnode=(struct stack*) malloc(sizeof(struct stack));
    printf("Enter the character\n");
    scanf("%c",&(newnode->ch));
    newnode->next=NULL;
    top=newnode;
    count++;
    }
    else
    {
    newnode=(struct stack*) malloc(sizeof(struct stack));
    printf("Enter the character\n");
    scanf("%c",&(newnode->ch));
    newnode->next=top;
    top=newnode;
    count++;
    }
    return count;
    }
    int pop(int count)
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    temp=top;
    printf("Deleted data on the stack is %c\n",temp->ch);
    top=top->next;
    free(temp);
    count--;
    }
    return count;
    }
    void peek()
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    printf("Data from the top of the stack is %c\n",top->ch);
    }
    }
    void isempty()
    {
    if(top==NULL)
    {
    printf("Stack is empty\n");
    }
    else
    {
    printf("Stack is not empty\n");
    }
    }
    void isfull()
    {
    newnode=(struct stack*) malloc(sizeof(struct stack));
    if(newnode==NULL)
    {
    printf("Stack is full\n");
    }
    else
    {
    printf("Stack is not full\n");
    }
    free(newnode);
    }
