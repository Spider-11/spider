#include<stdio.h>
int linearsearch(char str[],char key,int index)
{
	if(str[index]=='\0')
	{
		return -1;
	}
	if(str[index]==key)
	{
		return index+1;
	}
	linearsearch(str,key,index+1);
}
int main()
{
	char str[10];
	char key;
	printf("enter the string:");
	scanf("%s",str);
	printf("Enter the charrcter to search:");
	fflush(stdin);
	scanf("%c",&key);
	
	int result = linearsearch(str,key,0);
	if(result!=-1){
		printf("character %c found at index %d",key,result);
	}
	else
	{
		printf("character %c not found in the string",key);
	}
}
