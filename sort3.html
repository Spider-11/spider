#include<stdio.h>
    void bubble(int ar[],int size)
    {
    int i,j,temp,step=0;
    for(i=1;i<size;i++)
    {
    for(j=0;j<size-i;j++)
    {
    if(ar[j]>ar[j+1])
    {
    temp=ar[j];
    ar[j]=ar[j+1];
    ar[j+1]=temp;
    }
    }
    }
    }
