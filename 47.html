#include <stdio.h>
#include <string.h>
void textalignment(int alignment, char text[], int totalWidth) {
    int textLength = strlen(text);
    int i;
    if (alignment == 1) {
    
        printf("%s\n", text);
    } else if (alignment == 2) {
      
        int padding = (totalWidth - textLength) / 2;
        for ( i = 0; i < padding; i++) {
            putchar(' ');
        }
        printf("%s\n", text);
    } else if (alignment == 3) {
      
        int padding = totalWidth - textLength;
        for (i = 0; i < padding; i++) {
            putchar(' ');
        }
        printf("%s\n", text);
    } else {
        printf("Invalid alignment option\n");
    }
}
int main() {
    char text[20];
    int alignment;
    int totalWidth = 20;
    printf("Enter the text:");
 gets(text);
    printf("Choose alignment (1 for left, 2 for center, 3 for right): ");
    scanf("%d", &alignment);
    textalignment(alignment, text, totalWidth);
}
