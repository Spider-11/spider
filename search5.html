#include<stdio.h>
    int binary(int ar[],int item,int beg,int end)
    {
    int mid;
    while(beg<=end)
    {
    mid=(beg+end)/2;
    if(ar[mid]==item)
    {
    return mid;
    }
    if(ar[mid]>item)
    {
    end=mid-1;
    }
    else
    {
    beg=mid+1;
    }
    }
    return -1;
    }

    }

    {

    }
    }
