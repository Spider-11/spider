#include <stdio.h>
    int push(int item, int *top, char stack[], int size) {
        if (*top == size - 1) {
            printf("Stack overflow\n");
        } else {
            (*top)++;
            stack[*top] = item;
        }
        return *top;
    }

    int pop(char stack[], int *top) {
        if (*top == -1) {
            printf("Stack underflow\n");
            return -1;
        } else {
            return stack[(*top)--];
        }
    }

    int checkbracket(char expression[]) {
        int i, top = -1;
        char stack[10];

        for (i = 0; expression[i]; i++) {
            if (expression[i] == '(') {
                top = push('(', &top, stack, 9);
            } else if (expression[i] == ')') {
                if (pop(stack, &top) != '(') {
                    return -1;
                }
            }
        }

        if (top == -1) {
            return 1; 
        } else {
            return -1;
        }
    }

    int main() {
        char expression[10];
        int ch = -1;
        printf("Enter an expression with brackets: ");
        scanf("%s", expression);
        ch = checkbracket(expression);

        if (ch == 1) {
            printf("Brackets are balanced\n");
        } else {
            printf("Brackets are not balanced\n");
        }
    }
