#include <stdio.h>
int binarysearch(char str[], char item, int low, int high) {
    if (low > high) {
        return -1; 
    }
    int mid = (low + high) / 2;
    if (str[mid] == item) {
        return mid+1;
    } else if (str[mid] < item) {
        return binarysearch(str,item, mid + 1, high);  
    } else {
        return binarysearch(str,item, low, mid - 1);
    }
}
int main() {
    char str[10],item;
    int length;
    printf("Enter a  string: ");
    scanf("%s", str);
    printf("Enter a character to search: ");
    scanf(" %c", &item);
    length = 0;
    while (str[length] != '\0') {
        length++;
    }
    int result = binarysearch(str, item, 0, length - 1);
    if (result != -1) {
        printf("Character '%c' found at index %d\n", item, result);
    } else {
        printf("Character '%c' not found in the string\n", item);
    }
}
