#include <stdio.h>
#include <stdlib.h>
int push(char ch,char stack[],int size,int top) {
    if (top == size) {
	printf("Stack Overflow\n");
    } else {
	top++;
	stack[top]=ch;
	printf("Pushed \"%c\" onto the stack.\n", ch);
    }
    return top;
}

int pop(int top,char stack[]) {
    if (top == -1) {
	printf("Stack Underflow\n");
    } else {
	printf("Popped \"%c\" from the stack.\n", stack[top]);
	top--;
    }
    return top;
}

void peek(int top,char stack[]) {
    if (top == -1) {
	printf("Stack is empty. No element to peek.\n");
    } else {
	printf("Top element:\"%c\"\n", stack[top]);
    }
}

void isempty(int top) {
     if(top==-1)
     {
     printf("Stack is empty\n");
     }
     else
     {
     printf("Stack is not empty\n");
     }
}

void isfull(int top,int size) {
    if(top==size)
    {
    printf("Stack is full\n");
    }
    else
    {
    printf("Stack is not full\n");
    }
}
