#include<stdio.h>

void convert(int n) {
    if (n > 0) {
        convert(n / 2);
        printf("%d", n % 2);
    }
}

int main() {
    int num;

    printf("Enter a decimal number\t ");
    scanf("%d", &num);

    if (num < 0) {
        printf("Please enter a non-negative decimal number.\n");
    } else {

        printf("Binary representation: ");
        if (num == 0) {
            printf("0");
        } else {
            convert(num);
        }
        printf("\n");
    }
}
