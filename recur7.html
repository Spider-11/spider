#include<stdio.h>
int sum(int n) {

    if (n < 10) {
        return n;
    } else {
    
        return n % 10 + sum(n / 10);
    }
}
int main() {
    int num,sum2;
    printf("Enter a positive integer: ");
    scanf("%d", &num);
    if (num < 0) {
        printf("Enter a positive integer.\n");
    } else {
        sum2 = sum(num);
        printf("Sum of digits of %d = %d\n", num, sum2);
    }
}
