#include<stdio.h>
#include<string.h>
void selection(char str[][10],int size)
{
int i,j;
char temp[10];
for(i=0;i<size;i++)
{
for(j=i+1;j<size;j++)
{
if(strcmp(str[i],str[j])>0)
{
strcpy(temp,str[i]);
strcpy(str[i],str[j]);
strcpy(str[j],temp);
}
}
}
}
int main()
{
int i;
char str[5][10];
for(i=0;i<5;i++)
{
printf("Enter the strings at %d location\t",i+1);
scanf("%s",str[i]);
}
selection(str,5);
printf("Sorted strings are\n");
for(i=0;i<5;i++)
{
printf("%s\n",str[i]);
}
}
