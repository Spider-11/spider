#include <stdio.h>
int linearsearch(char str[], char item) {
    int index = 0;
    while (str[index] != '\0') {
        if (str[index] == item) {
            return index+1; 
        }
        index++;
    }
    return -1;  
}

int main() {
char str[50],item,result;
    printf("Enter a string: ");
       scanf("%s", str);
    printf("Enter a character to search: ");
    scanf(" %c", &item);
    result = linearsearch(str, item);
    if (result != -1) {
        printf("Character %c found at index %d", item, result);
    } else {
        printf("Character  not found in the string");
    }

}
