#include<stdio.h>
#include<string.h>
void bubble(char str[][10],int size)
{
int i,j;
char temp[10];
for(i=0;i<size-1;i++)
{
for(j=0;j<size-i-1;j++)
{
if(strcmp(str[j],str[j+1])>0)
{
strcpy(temp,str[j]);
strcpy(str[j],str[j+1]);
strcpy(str[j+1],temp);
}
}
}
}
int main()
{
char str[5][10];
int i;
printf("Enter the strings\n");
for(i=0;i<5;i++)
{
printf("Enter the string at location %d\t",i+1);
scanf("%s",str[i]);
}
bubble(str,5);
printf("Sorted Strings are\n");
for(i=0;i<5;i++)
{
printf("%s\n",str[i]);
}
}
