#include<stdio.h>
void quick(int *, int, int);
int partition(int *, int, int);

void quick(int a[], int low, int high) {
    int pivot;
    if (low < high) {
        pivot = partition(a, low, high);
        quick(a, low, pivot - 1);
        quick(a, pivot + 1, high);
    }
}

int partition(int a[], int low, int high) {
    int pivot, i, j, temp;
    pivot = a[high];
    i = low - 1;
    for (j = low; j < high; j++) {
        if (a[j] <= pivot) {
            i++;
            temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
    }
    temp = a[i + 1];
    a[i + 1] = a[high];
    a[high] = temp;
    return (i + 1);
}

    int i, a[10];

    }
    quick(a, 0,9);
    printf("Sorted array:");

        printf("%d\n", a[i]);

}
