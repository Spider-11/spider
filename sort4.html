#include<stdio.h>
int bubble(int ar[], int size, int index) {
	int j,temp;
    if (index == size - 1) {
        return 0;
    }
    for (j = 0; j < size - index - 1; j++) {
        if (ar[j] > ar[j + 1]) {
            int temp = ar[j];
            ar[j] = ar[j + 1];
            ar[j + 1] = temp;
        }
    }
    return bubble(ar, size, index + 1);
}
