#include<stdio.h>
    int push(int stack[],int top,int size)
    {
    int item;
    if(top==size)
    {
    printf("Stack is full\n");
    }
    else
    {
    printf("Enter the item you want to insert\n");
    scanf("%d",&item);
    top++;
    stack[top]=item;
    printf("%d is added in the stack\t",stack[top]);
    }
    return top;
    }
    int pop(int stack[],int top)
    {
    int item;
    if(top==-1)
    {
    printf("Stack is empty\n");
    }
    else
    {
    item=stack[top];
    printf("Deleted Item is %d\t",item);
    top--;
    }
    return top;
    }
    void peek(int stack[],int top)
    {
    int item;
    if(top==-1)
    {
    printf("Stack is empty\n");
    }
    else
    {
    item=stack[top];
    printf("Top element in the stack is %d\t",item);
    }
    }
    void isempty(int top)
    {
    if(top==-1)
    {
    printf("Stack is empty\n");
    }
    else
    {
    printf("Stack is not empty\n");
    }
    }
    void isfull(int top,int size)
    {
    if(top==size)
    {
    printf("Stack is full\n");
    }
    else
    {
    printf("Stack is not full\n");
    }
    }
