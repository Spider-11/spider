#include<stdio.h>

int sum(int n) {
    if (n == 1) {
        return 1;
    } else {
        return n + sum(n - 1);
    }
}

int main() {
    int result,num;
    printf("Enter the natural number\t");
    scanf("%d",&num);
    result=sum(num);
    printf("Sum of natural numbers %d\n", result);
}
