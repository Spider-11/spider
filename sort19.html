#include<stdio.h>
#include<string.h>
void swap(char str[][10], int a, int b) {
    char temp[10];
    strcpy(temp, str[a]);
    strcpy(str[a], str[b]);
    strcpy(str[b], temp);
}
int partition(char str[][10], int low, int high) {
    char pivot[10];
    int left,right;
    strcpy(pivot, str[low]);
    left = low + 1;
    right = high;

    while (1) {
        while (left <= right && strcmp(str[left], pivot) <= 0)
            left++;

        while (right >= left && strcmp(str[right], pivot) >= 0)
            right--;

        if (right < left)
            break;

        swap(str, left, right);
    }

    swap(str, low, right);
    return right;
}

void quicksort(char str[][10], int low, int high) {
    if (low < high) {
        int pivot = partition(str, low, high);
        quicksort(str, low, pivot - 1);
        quicksort(str, pivot + 1, high);
    }
}

int main() {
    int i;
    char str[5][10];
    printf("Enter the strings\n");
    for (i = 0; i < 5; i++) {
        printf("Enter the string at location %d: ", i + 1);
        scanf("%s", str[i]);
    }

    quicksort(str, 0, 4);

    printf("Sorted strings are:\n");
    for (i = 0; i < 5; i++) {
        printf("%s\n", str[i]);
}
}
